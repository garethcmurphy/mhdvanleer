215 template<class T>
216 Array2D<T> Array2D<T>::subarray(int i0, int i1, int j0, int j1) {
217  Array2D<T> A;
218  int m = i1 - i0 + 1;
219  int n = j1 - j0 + 1;
220 
221  /* if either length is zero or negative, this is an invalide
222  subarray. return a null view.
223  */
224  if (m < 1 || n < 1)
225  return A;
226 
227  A.data_ = data_;
228  A.m_ = m;
229  A.n_ = n;
230  A.v_ = Array1D<T *>(m);
231  T *p = &(data_[0]) + i0 * n_ + j0;
232  for (int i = 0; i < m; i++) {
233  A.v_[i] = p + i * n_;
234 
235  }
236  return A;
237 }
